### 定义
* 一种特殊线性表

### 特点
* 只允许在表尾删除、插入

### 示例
```
#!/usr/bin/python
# -*- coding: UTF-8 -*-

class StackTest(object):
    def __init__(self):
        self.items = []

    def is_empty(self):
        return self.items == []

    def push(self, item):
        self.items.append(item)

    def pop(self):
        self.items.pop();

    def peek(self):
        return self.items[len(self.items) - 1]

    def size(self):
        return len(self.items)


def reverseArrayItem(array):
    stackTest = StackTest()

    for item in array:
        stackTest.push(item)

    while stackTest.size() > 0:
        print(stackTest.peek())
        stackTest.pop()



if __name__ == "__main__":
    tmpArray = [1, 2, 3]
    reverseArrayItem(tmpArray)
```
